1.引入与应用
	import 文件名	引入一个文件
	from 文件名 import 类名 引用
2.特殊的类属性
	__name__	类的名字
	__doc__		文档的字符串
	__bases__	类的所有父类
	__dict__	类的所有属性(字典的形式)
	__module__	类所在的模块
	__class__	类型
3.特殊的实例属性
	__class__	实例所对应的类
	__dict__	实例所对应的属性
4.特殊的方法
	__init__()	初始化方法
	__new__()	构造器方法(在init之前,new方法中可以有return)
	__del__()	结构器
5.内置函数
	type		获取对象的类型
	isinstance	判断对象是不是一个类或者子类的实例
